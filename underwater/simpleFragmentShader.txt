#version 330 core

in vec3 LightIntensity;
in vec2 Texcoord; // 接收纹理坐标

out vec4 fragColor;

uniform sampler2D particleTexture; // 粒子纹理

void main() {
    vec4 textureColor = texture(particleTexture, Texcoord); // 从纹理中采样
    vec3 finalColor = LightIntensity * textureColor.rgb; // 将光照强度与纹理颜色结合
    fragColor = vec4(finalColor, textureColor.a); // 使用纹理的 alpha 值
}
