#version 330 core

in vec3 LightIntensity;
in vec2 Texcoord; // Received texture coordinates from the vertex shader

out vec4 fragColor;

uniform sampler2D objectTexture; // Object texture sampler (renamed for clarity)

void main() {
    // Sample color from the texture using Texcoord
    vec4 textureColor = texture(objectTexture, Texcoord);

    // Combine light intensity with the texture color
    vec3 finalColor = LightIntensity * textureColor.rgb;
    fragColor = vec4(finalColor, textureColor.a); // Use texture's alpha value
}
